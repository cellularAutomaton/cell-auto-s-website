This piece consists of 7 piano lines, each of which randomly moves up or down by fifths or fourths. Each part has a chance to skip a beat and then come in slightly early. Due to the desynchronization and to the drift around the circle of 5ths, the parts will disperse out over time. This is meant to be reminiscent of the 2nd law of thermodynamics. Video pending.`

from scamp import *
import random

s = Session(tempo=15)

s.fast_forward_in_beats(5000)
player = s.new_midi_part("midi through")


phrase_rhythms = [
    [1] * 2,
    [1.989893]
]

def random_improv(inst, start_pitch, skew=0.8, pitch_range=(44, 78)):
    pitch = start_pitch
    directionality = 1
    while True:
        for dur in random.choice(phrase_rhythms):
            volume = 1
            player.play_note(pitch, volume, dur)
            directionality *= random.choices([1, -1], weights=(skew, 1 - skew))[0]
            if pitch < pitch_range[0]:
                directionality = 1
            elif pitch > pitch_range[1]:
                directionality = -1
            pitch += directionality * random.choices([0, 5.0323, 6.9677])[0]

for i in range(7):
    fork(random_improv, args=(player, 62))
wait_for_children_to_finish()
